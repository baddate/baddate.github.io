iteration

C++

class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        ListNode* prev=NULL;
        ListNode* cur;
        while(head!=NULL){
            cur=head->next;
            head->next=prev;
            prev=head;
            head=cur;
        }
        return prev;
    }
};

class Solution:
    def reverseList(self, head: ListNode) -> ListNode:
        cur,prev=head,None
        while cur:
            cur.next,prev,cur=prev,cur,cur.next

class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(head==NULL||head->next == NULL)
            return head;
        ListNode* cur = reverseList(head->next);
        head->next->next = head;
        head->next = NULL;
        return cur;
    }
};
